 \subsection{Willekeurige driehoeken}
 \label{sec:wk_drhk}
  
\paragraph{Sinusregel}
\label{sec:Sinusregel}
 (waarbij $r$ de straal van de omgeschreven cirkel is)
   \[
	   \frac{a}{\sin \widehat{A}} = \frac{b}{\sin \widehat{B}} = \frac{c}{\sin \widehat{C}} = 2 \cdot r
   \]
