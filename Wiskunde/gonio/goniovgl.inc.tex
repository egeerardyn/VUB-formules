 \subsection{Vergelijkingen}
 \label{sec:goniovgl}
  \[
	   \sin x = \sin \alpha
	   \Leftrightarrow 
	   x = \alpha + k \cdot 2\pi \vee x = \pi - \alpha + k \cdot 2\pi  
	   \qquad
	   \textrm{ met } k \in \mathbb{Z}
  \]
  \[
	   \cos x = \cos \alpha 
	   \Leftrightarrow
	   x = \alpha + k \cdot 2\pi \vee x = - \alpha + k \cdot 2\pi
	   \qquad
	   \textrm{ met } k \in \mathbb{Z}
  \]
  \[
	   \tan x = \tan \alpha 
	   \Leftrightarrow
	   x = \alpha + k \cdot \pi 
	   \qquad
	   \textrm{ met } k \in \mathbb{Z}
  \]
  \[
	   \cot x = \cot \alpha 
	   \Leftrightarrow
	   x = \alpha + k \cdot \pi 
	   \qquad
	   \textrm{ met } k \in \mathbb{Z}
  \]
